- [ ] I = insert start of line
- [ ] a = insert after cursor
- [ ] A = insert end of line
- [ ] o = start next new line

- [ ] w = move forward 1-word (start of word)
- [ ] e = move forward 1-word (end of word)
- [ ] b = move backward 1-word 
- [ ] 0 = move to start of line
- [ ] ^ = move to first character of line
- [ ] $ = move to end of line

- [ ] f + CHAR = find next instance of CHAR 
- [ ] F + CHAR = find last instance of CHAR 
- [ ] ; = repeat the find forward (in same line) 
- [ ] , = repeat the find backwards (in same line) 

- [ ] == = ?fix indentiation of whole line

- [ ] gg = move to first line 
- [ ] G = move to last line 
- [ ] M = move to middle line 
- [ ] zz = scroll so cursor is middle line 

- [ ] dw = delete word / cut 
- [ ] dd = delete line / cut 
- [ ] u = undo 
- [ ] CRTL+r = redo 

- [ ] yw = yank word
- [ ] yy + p = yank current line + paste next line 

- [ ] % = Connected pair (e.g. get to closing bracket) 

- [ ] diw = delete-in word 
- [ ] daw = delete-around word 
- [ ] yiw = yank-in word 
- [ ] di( = delete-in parenthesis (or any enclosing pair - while inside the pair)
- [ ] ciw = change inner word (current word) 
- [ ] ci( = change inner word (current word) 

- [ ] r = replace current char
- [ ] . = repeat last action 

- [ ] / = start search forward
- [ ] ? = start search backwards
- [ ] n = next search result
- [ ] N = last search result

- [ ] :s/old/new = Replace first occurrence of "old"
- [ ] :%s/old/new/g = Replace all occurrences of "old"
